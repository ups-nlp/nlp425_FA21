def grabTree(self):
        tree = open(r"tree.txt","w+")
        tree.write(self.root.treeStr(0))
        tree.close()

def treeStr(self,level):
        s = ""
        space = ""
        for i in range(level):
            space +=">"
        if(self.prevAct is None):
            s += "\t"+space+"<root>"+"\n"
        else:
            s+= "\t"+space+self.prevAct+"\n"
        for child in self.children:
            s+=child.treeStr(level+1)
        return s
